function lightbox(insertContent, ajaxContentUrl){
	// remove any previously added content
	$('#lightbox').empty();

	// insert HTML content
	if(insertContent != null){
		$('#lightbox').append(insertContent);
	}

	// insert AJAX content
	if(ajaxContentUrl != null){
		// temporarily add a "Loading..." message in the lightbox
		$('#lightbox').append('<p class="loading">Loading...</p>');

		// request AJAX content
		$.ajax({
			type: 'GET',
			url: ajaxContentUrl,
			success:function(data){
				// remove "Loading..." message and append AJAX content
				$('#lightbox').empty();
				$('#lightbox').append(data);
			},
			error:function(){
				alert('팝업 로드 실패!');
			}
		});
	}

	// display the lightbox

	$('#lightbox_wrap').show(500, 	
		function(){
			if($("#lightbox_wrap").height() > window.innerHeight){
				$("#lightbox_wrap").removeClass("pop_middle");
				$("#lightbox_wrap").addClass("pop_top");
			}
			else if($("#lightbox_wrap").height() < window.innerHeight) {
				$("#lightbox_wrap").removeClass("pop_top");
				$("#lightbox_wrap").addClass("pop_middle");
			}
		}
	);
	$('#closepop_wrap').show(500);
	$('#lightbox-shadow').show();
	$('body').addClass('document_overflow');
}

	function event_IsLogin() {
        var islogin = false;
        $.ajax({
            url: '/Ajax/IsLoginCheck',
            type: 'POST',

            dataType: "json",
            async: false,
            success: function (res) {
                islogin = (res.return == "1");

            error: function (request, status, error) {
                console.log("code:" + request.status + "\n" + "message:" + request.responseText + "\n" +
                    "error:" + error);
            }
        });

    }

    function event_IssueEventLanding() {

			type: 'POST',

			contentType: "application/x-www-form-urlencoded; charset=utf-8",

	function event_SetSlide(info){
		$('.coming_btn').each(function(index,obj){
			if(index > parseInt(info.PRIZE_DESC)){
				$(this).show();
			}
		});

			if(index < parseInt(info.PRIZE_DESC)){

		}

	}
	function event_SetAgree(obj){
		if(!event_IsLogin())
        {
            if (confirm("로그인 후 이용 가능합니다.\n로그인 하시겠습니까?")) {
                WCK_COMMON_RENEWAL_FN.getLogin();
            }  
        }
		else{
			var chkcnt = 0;
			$('.rozy_applywrap').find('input:checkbox').each(function(){
				if(!$(this).is(':checked') && $(this).attr('id') != 'mar_all_checkbtn' && $(this).attr('id') != 'agreeall_checkbtn'){
					chkcnt++;
				}
			});

			if(chkcnt > 0){
				alert('개인정보 및 모든 마케팅 정보 수신에 동의해주세요.');
				return;
			}

			if(!isProcess){
				isProcess = true;
				$.ajax({
					url: '/Event/SetEventAgreeReceiveMarketing',
						data: {"types": 'S,E'},
						type: 'POST',
						cache: false,
						contentType: "application/x-www-form-urlencoded; charset=utf-8",
						dataType: "json",
						async: false,
						success: function (result) {

						}
				});
			}
		}
	}
	function event_Entry(obj){
		if (!event_IsLogin())
        {
            if (confirm("로그인 후 이용 가능합니다.\n로그인 하시겠습니까?")) {
                WCK_COMMON_RENEWAL_FN.getLogin();
            }  
        }

        if (isProcess == false) {
			isProcess = true;
			var seq = $(obj).attr("data-seq");

            $.ajax({
                url: '/Event/IssueEventEntry',
                data: {
                    issueNo: 17051,
                    promotionMaster: 160,
                    promotionPrizeMaster: seq
                },
                type: 'POST',
                cache: false,
                contentType: "application/x-www-form-urlencoded; charset=utf-8",
                dataType: "json",
                async: false,
                success: function (res) {
                    isProcess = false;

                    var result = res[0];
                    if (result.resultcode != undefined) {
                        var resultcode = result.resultcode;
                        if (resultcode == -5) {
                            if (confirm("로그인 후 이용 가능합니다."))
                                WCK_COMMON_RENEWAL_FN.getLogin();
                            return;
                        } else if (resultcode == 6) {
                            alert("본인 인증을 진행해주세요.");
                            return;
                        } else if (resultcode == 98) {
							alert("오늘의 럭키드로우 응모가 완료되었습니다. 내일 또 만나요!");
							$(obj).hide();
							$(obj).next().show();
                            return;
                        } else if (resultcode == 0) {
							if(islastday == "Y"){
								alert("오늘의 럭키드로우 응모가 완료되었습니다. 당첨자는 11/18에 발표됩니다.");
							}else{
								alert("오늘의 럭키드로우 응모가 완료되었습니다. 내일 또 만나요!");
							}
							$(obj).hide();
							$(obj).next().show();
                            return;
                        } else if (resultcode == 77 || resultcode == 88) {
                            alert("이벤트 기간이 아닙니다.");
                            return;
                        } else {
                            alert('다시 시도 하세요.');
                            return;
                        }
                    }
                },
                error: function (request, status, error) {
                    alert("error: 다시 시도 하세요.");
                    console.log("code:" + request.status + "\n" + "message:" + request.responseText + "\n" + "error:" + error);
                },
                complete: function (XMLHttpRequest, textStatus) {
                    isProcess = false;
                }
            });
            
        }
        else {
            alert("처리중입니다. 잠시만 기다려주세요.");
        }
	}
