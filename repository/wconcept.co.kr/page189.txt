     function FindSizePop() {

            $.ajax({
                type: "POST",
                url: "/Product/CustomerRecomSizeMember",

                        var height = result[0].HEIGHT;
                        var weight = result[0].WEIGHT;
                        var recommsize = result[0].SIZE;

                        $('.size_label input[name="sizeRadio1"]').each(function (index) {
                            var res = $(this).val().replaceAll(" ", "").replace("cm", "").replace("이하", "").replace("이상", "");

                                $(this).trigger("click");
                                $(this).prop("checked", true);

                            var res = $(this).val().replaceAll(" ", "").replace("kg", "").replace("이하", "").replace("이상", "");

        function FundingEndTimer(distance) {
            var day, hour, min, sec;

            distance  = distance  - 1000;

            day =  Math.floor(distance  / (1000 * 60 * 60 * 24));//일
            hour = Math.floor((distance  / (1000*60*60)) % 24);//시간
            min = Math.floor((distance / (1000 * 60)) % 60);//분
            sec = Math.round((distance / 1000) % 60);

            if (day < 10) {
                day = "0" + String(day);
            }
            if (hour < 10) {
                hour = "0" + String(hour);
            }
            if (min < 10) {
                min = "0" + String(min);
            }
            if (sec < 10) {
                sec = "0" + String(sec);
            }

            if (distance < 0) {
                clearInterval(timerId);
                $('.fundingItemPre > button').removeClass('black').addClass('right_gray').text('SOLD OUT').prop('onclick', null);
            }
            else {
                var text = stringformat('<span class="day">{0}</span>일&nbsp;<span class="hour">{1}</span>시간&nbsp;<span class="minute">{2}</span>분&nbsp;<span class="minute">{3}</span>초&nbsp;남음', day, hour, min, sec);
                $('#funding_day').html(text);
            }

            var timerId = setTimeout('FundingEndTimer(' + distance + ')', 1000);
        }
