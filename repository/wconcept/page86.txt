     var makeTimer174310 = setInterval(function () {
        var sectionId = "CountDown174310";

        var now = new Date(); //현재시간
        var startTime = Date.parse("2022/05/10 10:00:00");       //시작일시
        var endTime = Date.parse("2022/05/20 18:00:59") + 1000;   //종료일시 (+1초로 종료날짜 셋팅)
        var distance = endTime - (now > startTime ? now : startTime);       //시작날짜가 현재시간보다 큰 경우 Timer Start

        //Coming soon제거 (카운트다운 1초 전에 해제)
        if (now >= (startTime - 1000) && $("#" + sectionId).hasClass("st_coming_soon") === true) {
            $("#" + sectionId).removeClass("st_coming_soon");
        }

        var days = Math.floor(distance / (1000 * 60 * 60 * 24));
        var hours = Math.floor(distance / (1000 * 60 * 60));
        var minutes = Math.floor(distance / (1000 * 60));
        var seconds = Math.floor(distance / 1000);

        var d = days < 0 ? 0 : days;
        var h = hours < 0 ? 0 : hours - (days * 24);
        var m = minutes < 0 ? 0 : minutes - (hours * 60);
        var s = seconds < 0 ? 0 : seconds - (minutes * 60);

        if (d < 1) { d = "0" + d; }
        if (h < "10") { h = "0" + h; }
        if (m < "10") { m = "0" + m; }
        if (s < "10") { s = "0" + s; }

        $("#" + sectionId + " .time_day>.digit").text(d);
        $("#" + sectionId + " .time_hour>.digit").text(h);
        $("#" + sectionId + " .time_min>.digit").text(m);
        $("#" + sectionId + " .time_sec>.digit").text(s);

        //날짜비노출 시간만으로 24시간이상 설정시
        if (days >= 1 && $("#" + sectionId + " .countdown_timer").hasClass("ty_hour_only") === true) {
            $("#" + sectionId + " .ty_hour_only > .time_hour>.digit").text(hours);
        }

        //시간을 분으로 설정 시
        if (hours >= 1 && $("#" + sectionId + " .countdown_timer").hasClass("ty_minutes_only") === true) {
            $("#" + sectionId + " .ty_minutes_only > .time_min>.digit").text(minutes);
        }

        //0 이되면 멈춤
        if (seconds < 1) {
            clearInterval(makeTimer174310);
        }
    }, 1000);
