var calcs = document.querySelectorAll('.js-calculator')
var calcsNav = document.querySelectorAll('.js-calculator-nav')

function changeCalc(hash) {
    var calcId = (hash || window.location.hash).replace('#', '')
    var calcNav = document.querySelector('a[href="#' + calcId + '"]')
    var calcsActive = document.getElementById(calcId)

    if (calcsActive) {
        resetCalcNav()
        calcNav.classList.add('is-selected')
        calcsActive.classList.remove('d-none')
    }
}

function resetCalcNav() {
    for (var i = 0, calc; calc = calcs[i]; i++) {
        calc.classList.add('d-none')
    }
    for (var i = 0, nav; nav = calcsNav[i]; i++) {
        nav.classList.remove('is-selected')
    }
}

for (var i = 0, nav; nav = calcsNav[i]; i++) {
    nav.addEventListener('click', function(e) {
        e.preventDefault()
        var hash = e.target.getAttribute('href')
        
        if (window.history.pushState) {
            window.history.pushState(null, null, hash)
        }
        else {
            location.hash = hash
        }

        changeCalc(hash)
    })
}

/* Query strings and url setting */

function updateQs(fields) {
    var qs = new URLSearchParams()

    for (var field in fields) {
        qs.set(field, fields[field])
    }

    var newLocation = '?' + qs.toString()
    window.history.replaceState({}, document.title, newLocation + window.location.hash)

    var shareInput = document.querySelector('input[name="teamsROIReportURL"]')

    if (shareInput) {
        shareInput.value = window.location.href
    }
}

function getFieldValues(evt) {
    var allFields = {}
    var enabledFields = {}

    // used to be getFieldValues(elements)
    // elements was passed in the function but we want to save all fields, not just the active form
    var elements = document.querySelectorAll('.js-calc-input')

    for (var i = 0, field; field = elements[i]; i++) {
        var val = parseInt(document.getElementById(field.id).value) || 0
        
        allFields[field.id] = val
        if (!field.disabled) enabledFields[field.id] = val
    }

    if (evt && evt.target) {
        updateQs(enabledFields)
    }

    return allFields
}

function updateFieldsFromQs() {
    var qs = new URLSearchParams(window.location.search)

    for (let field of qs) {
        var elm = document.getElementById(field[0])
        if (elm && !elm.disabled) {
            document.getElementById(field[0]).value = field[1]
        }
    }
}
