        var filterTimeout = null;

        $(".js-filter").keyup(function () {
            clearTimeout(filterTimeout);
            filterTimeout = setTimeout(filter, 300);
        });

        function castVote(votetype, synonymId, $tr) {
            $.ajax({
                url: '/tag-synonym/' + votetype + 'vote/' + synonymId,
                data: { fkey: StackExchange.options.user.fkey },
                error: function (request, message) {
                    StackExchange.helpers.showToast(request.responseText, { type: 'danger' });
                },
                success: function (result) {
                    if (result.approved) {
                        $tr.fadeOutAndRemove();
                        StackExchange.helpers.showToast('Successfully approved the proposed synonym.', { type: 'success' });
                    } else if (result.rejected) {
                        $tr.fadeOutAndRemove();
                        StackExchange.helpers.showToast('Successfully rejected the proposed synonym.', { type: 'success' });
                    } else {
                        $tr.find('.js-synonym-score').text(result.score);
                        var $upvote = $tr.find('.js-synonym-upvote');
                        var $downvote = $tr.find('.js-synonym-downvote');
                        if (result.upvoted) {
                            $upvote.removeClass('fc-black-150').addClass('fc-theme-primary js-voted');
                            $upvote.attr('title', 'Undo upvote');
                        } else {
                            $upvote.addClass('fc-black-150').removeClass('fc-theme-primary js-voted');
                            $upvote.attr('title', 'Vote up this synonym');
                        }
                        if (result.downvoted) {
                            $downvote.removeClass('fc-black-150').addClass('fc-theme-primary js-voted');
                            $downvote.attr('title', 'Undo downvote');
                        } else {
                            $downvote.addClass('fc-black-150').removeClass('fc-theme-primary js-voted');
                            $downvote.attr('title', 'Vote down this synonym');
                        }
                    }
                },
                type: 'POST',
                dataType: 'json'
            });
        }

        $(document).on('click', '.js-synonym-upvote, .js-synonym-downvote', function () {
            var $this = $(this);
            var $tr = $(this).closest('tr');
            var isUpvote = $this.hasClass('js-synonym-upvote');

            var threshold = parseInt($this.data('threshold'));
            var scoreStr = $(this).closest('tr').find('.js-synonym-score').text();
            var currentScore = parseInt(scoreStr);

            var showModal = !$this.hasClass('js-voted') && ((isUpvote && currentScore + 1 === threshold) || (!isUpvote && currentScore - 1 === threshold));

            var votetype = isUpvote ? 'up' : 'down';
            var synonymId = parseInt($tr.attr('class').match(/\d+/)[0], 10);

            if (!showModal) {
                castVote(votetype, synonymId, $tr);
            } else {
                var $modal = $('.js-final-vote-confirmation-modal').clone(false);

                if (isUpvote) {
                    $modal.find('.js-final-vote-modal-description').html('Once you cast this vote, this synonym will be approved. This vote cannot be undone.');
                } else {
                    $modal.find('.js-final-vote-modal-description').html('Once you cast this vote, this synonym will be rejected. This vote cannot be undone.');
                }

                $modal.find('.js-submit').on('click', function (e) {
                    e.preventDefault();
                    e.stopPropagation();

                    castVote(votetype, synonymId, $tr);
                    $modal.fadeOutAndRemove();
                });

                StackExchange.helpers.showModal($modal, { returnElements: $this });
            }

            return false;
        });

        function positionPopover($popover, $source, offset) {
            var sourcePosition = $source.position();
            var pW = $popover.outerWidth();

            offset = offset || 0;

            var popoverLeft = (sourcePosition.left * 2 + $source.outerWidth()) / 2 - pW / 2 - offset;
            var popoverTop = sourcePosition.top + $source.outerHeight() + 12 - offset;

            $popover.css({
                position: 'absolute',
                top: (popoverTop | 0) + "px",
                left: (popoverLeft | 0) + "px"
            });
        };

        var scoreTimeout = null;
        $(document).on('mouseenter', '.js-synonym-score', function (e) {
            var $popover = $('.js-popover-score');
            var currentlyOpen = $popover.hasClass('is-visible');

            positionPopover($popover, $(this), 12);

            if (!currentlyOpen) {
                clearTimeout(scoreTimeout);
                scoreTimeout = setTimeout(function () {
                    $popover.addClass("is-visible");
                }, 300);
            }

            e.preventDefault();
            e.stopPropagation();
        });
        $(document).on('mouseleave', '.js-synonym-score', function (e) {
            clearTimeout(scoreTimeout);
            $('.js-popover-score').removeClass('is-visible');
            e.preventDefault();
            e.stopPropagation();
        });

    
    });
