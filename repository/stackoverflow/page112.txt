      StackExchange.ready(function () {
        //functions for updating Post attributes filter elements
        function checkBoxes(){
            $(".js-filter-attribute-btn").each(function () {
                if($(this).val() == "All")
                {
                    if(filterSelected("All"))
                    {
                        $(this).addClass("d-none");
                    }
                }
                else if(filterSelected($(this).val()))
                {
                    $(this).prop("checked", true);
                }
                else
                {
                    $(this).prop("checked", false);
                }
            });
        };

        function countAttributes(){
            if($("#js-post-attribute").val() == "All" || $("#js-post-attribute").val() == "") 
            {
                $("#js-clear-types-btn").addClass("d-none");
                $("#js-attributecount").addClass("d-none");
            } 
            else 
            {
                $("#js-clear-types-btn").removeClass("d-none");
                $("#js-attributecount").removeClass("d-none");
                $("#js-attributecount > span.s-btn--number").text($("#js-post-attribute").val().split(" ").length);
            }
        }

        function disableCheckboxes(){
            $(".js-filter-attribute-btn").each(function () {
                let toBeDisabled = "";

                if(filterSelected("Unanswered")) 
                {
                    toBeDisabled += "Answered Accepted Recommended Unaccepted Unrecommended";
                }
                if(filterSelected("Answered")) 
                {
                    toBeDisabled += "Unanswered";
                }
                if(filterSelected("Unaccepted")) 
                {
                    toBeDisabled += "Accepted";
                }
                if(filterSelected("Accepted")) 
                {
                    toBeDisabled += "Unanswered Unaccepted";
                }
                if(filterSelected("Unrecommended")) 
                {
                    toBeDisabled += "Recommended";
                }
                if(filterSelected("Recommended")) 
                {
                    toBeDisabled += "Unanswered Unrecommended";
                }
                           

                if(toBeDisabled.indexOf($(this).val()) >= 0)
                {
                    $(this).attr("disabled", true);
                    $(this).parent('div').addClass("is-disabled");
                }
                else
                {
                    $(this).removeAttr("disabled");
                    $(this).parent('div').removeClass("is-disabled");
                }
            });
        };

        function filterSelected(str){
            return ($("#js-post-attribute").val().indexOf( str ) >= 0);
        };

        function removeFilter(e){
            switch(e.data("filtertype")) 
            {
                case "tag":
                        var tagInput = $("input[name='filter-tags']");
                        var newVal = tagInput.val().replace(e.data("filtername"), "").replace("  ", " ").trim();
                    tagInput.val(newVal);
                    break;
                case "daterange":
                    var dateRangeInput = $("select[name='date-range-value']");
                    dateRangeInput.find("option:selected").prop("selected", false);
                    dateRangeInput.find("option:disabled").removeAttr("disabled").prop("selected", true);
                    break;
                case "attribute":
                    var attributeInput = $("#js-post-attribute");
                    var newVal = attributeInput.val().replace(e.data("filtername"), "").replace("  ", " ").trim();
                    newVal = newVal == "" ? "All" : newVal;
                    attributeInput.val(newVal);
                    break;
                default:
                    break;
            };
            $("#js-submit-filters").click();
        };

        function updatePostAttributesElements(){
            countAttributes()
            checkBoxes();
            disableCheckboxes();
        };

        function updateTagCount(val) {
            var newTagCount = (val == "") ? 0 : val.split(" ").length;
            if(newTagCount == 0)
            {
                $("#js-tagcount").addClass("d-none");
            } 
            else 
            {
                $("#js-tagcount").removeClass("d-none");
            }
            $("#js-tagcount > span.s-btn--number").text(newTagCount);
        }

        //General filter events
        $(".js-post-sort").on("change", function (e) {
            e.preventDefault();
            e.stopPropagation();

            var nav = $(e.target).find(":selected").data("nav-value");
            var sortBtn = $("#js-sort-submit");
            sortBtn.attr("data-nav-value", nav);

            sortBtn.attr("href", e.target.value);

            sortBtn[0].click();
        });

        $(".js-filter-dismiss").on("click", function (e) {
            e.preventDefault();
            removeFilter($(this));
        });

        $(".js-filter-dismiss").on("keypress", function (e) {
            if(e.keyCode == 13) //enter key
            {
                e.preventDefault();
                removeFilter($(this));
            }
        });

        $("#js-clear-btn").on("click", function () {
            $("#submit-reset").click();
        });

        $(".js-post-filter-cancel-btn").on("click", function (e) {
            e.preventDefault();
            e.stopPropagation();

            //reset filter inputs
            var tagInput = $("input[name='filter-tags']");
            tagInput.val($(this).data("tag-recall"));
            tagInput.next().hide();
            tagInput.show();
            updateTagCount(tagInput.val());

            $("#js-post-attribute").val($(this).data("attribute-recall"));
            updatePostAttributesElements();

            $("select[name='date-range-value']").val($(this).data("date-range-recall"));

            $(".js-post-filter-toggle-btn").click();
        });

        //Tags filter events
        $(".js-filter-tag-btn").on("click", function (e) {
            e.preventDefault();
            e.stopPropagation();
            StackExchange.using("tagEditor", () => {
                initTagRenderer([""], [""]);
                StackExchange.tagEditor(
                    $("input[id $= filter-tags]"),
                    {
                        excerpts: false,
                        useStacksClasses: true,
                        columns: 1,
                        suggestionsTopMargin: 10
                    }
                );
            });

            $("#post-tags").on('keypress', function (e) {
                if(e.keyCode == 13) //enter key
                {
                    e.preventDefault();
                    $("#js-submit-filters").click();
                }
            });

            $("input[name='filter-tags']").on("change", function (e) {
                updateTagCount(e.target.value);
            });
        });


        //Post attributes filter events
        $(".js-filter-attribute-btn-group").on("click", function () {
            checkBoxes();
            disableCheckboxes();
        });

        $(".js-filter-attribute-btn").on("click", function (e) {
                    
            if (filterSelected($(this).val())) 
            { //if filter was already selected, remove that filter
                var newVal = $("#js-post-attribute").val().replace($(this).val(), "").replace("  ", " ").trim();
                $("#js-post-attribute").val(newVal);
            } else 
            { //else, add the new filter and remove "All" and show Clear button
                var newVal = (`${$("#js-post-attribute").val()} ${$(this).val()}`).replace("  ", " ").trim();
                if (filterSelected("All"))
                {
                    newVal = newVal.replace("All", "").trim();
                }
                        
                $("#js-post-attribute").val(newVal);
            }

            if($(this).val() == "All" || $("#js-post-attribute").val() == "")
            { //If Clear button was clicked or filter is empty, default to "All"
                $("#js-post-attribute").val("All");
            }

            updatePostAttributesElements();
        });

        $(".js-filter-attribute-btn").on('keypress', function (e) {
            if(e.keyCode == 13) //enter key
            {
                e.preventDefault();
            }
        });

    });
